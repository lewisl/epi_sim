// states.cpp
#include "states.h"
#include <cstdlib>

namespace {
    States g_states;
    bool g_inited = false;
}

void init_states(States s) {
    if (g_inited) std::abort();   // optional: enforce "only once"
    g_states = s;
    g_inited = true;
}

const States& states() {
    if (!g_inited) std::abort();  // optional: enforce "must init first"
    return g_states;
}
